## Create / Clone Repo
git init     
git clone [repo-url]

## Staging and committing
git status
git add .
git commit - m "commit changes"

## Branching and Merging
git branch 
git merge [branch name]

## For remote repo
git remote -v
git remote add origin (repo url)
git remote remove origin

## Push and Pull
git push origin main
git pull origin main

## Fetch origin
git fetch origin
